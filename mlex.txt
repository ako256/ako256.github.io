import React, { useState } from "react";
import "./App.css"; // or "./form.css" if you prefer

export default function UserInfoForm() {
  const [userData, setUserData] = useState({
    name: "",
    age: "",
    location: "",
    hobby: "",
  });

  const [submittedData, setSubmittedData] = useState(null);

  // âœ… Handle input change
  const handleChange = (e) => {
    const { name, value } = e.target;
    setUserData((prev) => ({
      ...prev,
      [name]: value,
    }));
  };

  // âœ… Handle submit with in-page JS alerts
  const handleSubmit = (e) => {
    e.preventDefault();

    if (!userData.name.trim()) {
      alert("Please enter your full name!");
      return;
    }

    if (!userData.age) {
      alert("Please enter your age!");
      return;
    } else if (userData.age < 1 || userData.age > 120) {
      alert("Age must be between 1 and 120!");
      return;
    }

    if (!userData.location.trim()) {
      alert("Please enter your location!");
      return;
    }

    if (!userData.hobby.trim()) {
      alert("Please enter your hobby!");
      return;
    }

    // âœ… If all fields are filled correctly
    setSubmittedData(userData);
    alert("Data submitted successfully!");

    // Reset form
    setUserData({
      name: "",
      age: "",
      location: "",
      hobby: "",
    });
  };

  return (
    <div className="info-root">
      <h1 className="info-title">User Information Form</h1>

      <form className="info-form" onSubmit={handleSubmit}>
        <label className="info-label" htmlFor="name">Name:</label>

          placeholder="Enter your full name"
          value={userData.name}

http-server.js

const http = require('http');
const fs = require('fs');
const url = require('url');
const path = require('path');


// Create the server
const server = http.createServer((req, res) => {
  let pathname = url.parse(req.url).pathname;


  console.log(`Request for ${pathname} received`);


  // Default to 'First.html' if the root is accessed
  if (pathname == '/') {
    pathname = '/First.html';  // Default to First.html
  }


  // Dynamically resolve the file path based on request
  let filePath = path.join(__dirname, pathname);  // __dirname gives the current directory


  // Print the full path to debug
  console.log(`File path requested: ${filePath}`);


  // Check if the file exists and serve it
  fs.readFile(filePath, (err, data) => {
    if (err) {
      console.error(err);


      // Send 404 if the file is not found
      res.writeHead(404, { 'Content-Type': 'text/plain' });
      res.write('404 - file not found');
    } else {
      // Serve the requested file
      res.writeHead(200, { 'Content-Type': 'text/html' });
      res.write(data.toString());
    }


    res.end();
  });
});


// Start the server
server.listen(8001, () => {
  console.log('Server running on port 8001');
});

App.js
import React, { useState, useEffect } from 'react';

const API = 'http://localhost:3000/employees';

function App() {
  const [employees, setEmployees] = useState([]);
  const [form, setForm] = useState({ name: '', age: '', city: '' });

  const fetchEmployees = async () => {
    const res = await fetch(API);
    const data = await res.json();
    setEmployees(data);
  };

  useEffect(() => { fetchEmployees(); }, []);

  const handleChange = e => setForm({ ...form, [e.target.name]: e.target.value });

  const handleSubmit = async e => {
    e.preventDefault();
    await fetch(API, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ ...form, age: Number(form.age) })
    });
    setForm({ name: '', age: '', city: '' });
    fetchEmployees();
  };

  const handleDelete = async id => {
    await fetch(`${API}/${id}`, { method: 'DELETE' });
    fetchEmployees();
  };

  return (
    <div style={{ margin: '2rem' }}>
      <h1>Employee CRUD</h1>
      <form onSubmit={handleSubmit}>
        <input name="name" placeholder="Name" value={form.name} onChange={handleChange} />
        <input name="age" type="number" placeholder="Age" value={form.age} onChange={handleChange} />
        <input name="city" placeholder="City" value={form.city} onChange={handleChange} />
        <button type="submit">Add Employee</button>
      </form>
      <ul>
        {employees.map(emp => (
          <li key={emp._id}>
            {emp.name} ({emp.age}) - {emp.city}
            <button onClick={() => handleDelete(emp._id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

export default App;

employeeController.js
const { ObjectId } = require('mongodb');
const { getCollection } = require('./employeeModel');

async function getEmployees(req, res) {
  try {
    const col = await getCollection();
    const employees = await col.find({}).toArray();
    res.json(employees);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
}

async function createEmployee(req, res) {
  try {
    const col = await getCollection();
    const result = await col.insertOne(req.body);
    res.status(201).json({ insertedId: result.insertedId });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
}

async function updateEmployee(req, res) {
  try {
    const { id } = req.params;
    const col = await getCollection();
    const result = await col.updateOne({ _id: new ObjectId(id) }, { $set: req.body });
    res.json(result);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
}

async function deleteEmployee(req, res) {
  try {
    const { id } = req.params;
    const col = await getCollection();
    const result = await col.deleteOne({ _id: new ObjectId(id) });
    res.json(result);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
}

module.exports = { getEmployees, createEmployee, updateEmployee, deleteEmployee };
